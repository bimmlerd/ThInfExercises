\begin{homeworkProblem}
	Wir suchen \( \bigcup_{k \in {0,1,2,3}} R_{13}^k \) \\
	Es ist offensichtlich, dass:  \\
	\begin{tabular}{|r|l|}
		\hline
		\(R_{11}^0 \) & \(c\) \\
		\(R_{12}^0 \) & \(a + b\) \\
		\(R_{13}^0 \) & \( \emptyset\) \\
		\(R_{21}^0 \) & \( \emptyset\) \\
		\(R_{22}^0 \) & \( a + c\) \\
		\(R_{23}^0 \) & \( b\) \\
		\(R_{31}^0 \) & \( \emptyset\) \\
		\(R_{32}^0 \) & \( c\) \\
		\(R_{33}^0 \) & \( a +b \) \\
		\hline
	\end{tabular}
	Mit Hilfe der Formel \( R_{ij} = R_{ij}^{k-1} + R_{ik}(R_{kk})^{*}R_{kj} \) können wir nun \(R_{13}^(k) \) für 				aufsteigende k berechnen.\\
	\begin{enumerate}[-]
	\item \(k = 1\)
		\begin{tabular}[|l|c|r|]
			& direktes Einfügen & vereinfacht \\
			\hline \hline
			\(R_{11}^1 \) & \(c + cc^{*}c \) & \( c^{*} \) \\
			\(R_{12}^1 \) & \( (a + b) + c^{*} (a + b) \) & \(c^{*}(a+b) \) \\
			\(R_{13}^1 \) & \( \emptyset + cc^{*} \emptyset\) & \( \emptyset\) \\
			\(R_{21}^1 \) & \( \emptyset +  \emptyset c^{*}(a+b)\) & \(\emptyset\) \\
			\(R_{22}^1 \) & \( (a+c) + \emptyset c^{*} (a+b)\) & \(\a+\centering\) \\
			\(R_{23}^1 \) & \(  b + \emptyset c{*}(a+b)\) & \( b\)     \\
			\(R_{31}^1 \) & \( \emptyset + \emptyset c{*}\) & \( \emptyset\) \\
			\(R_{32}^1 \) & \( c + \emptyset c{*} (a+b)\) & \( c \)  \\
			\(R_{33}^1 \) & \( (a +b) + \emptyset c{*} \emptyset \) & \( a + b \) \\
			\hline
		\end{tabular}
	\item \(k = 2\)
		\begin {tabular}[|l|c|r|]
			& direktes Einfügen & vereinfacht \\
			\hline \hline
			\(R_{11}^2 \) & \(c^{*} + c^{*}(a+b)(a+c)^{*} \emptyset \) & \( c^{*} \) \\
			\(R_{12}^2 \) & \( c^{*}(a + b) + c^{*} (a + b)(a+c)^{*} \) & \(c^{*}(a+b)(a+c)^{*} \) \\
			\(R_{13}^2 \) & \( \emptyset + c^{*}(a+b)(a+c)^{*}b) & \( c^{*}(a+b)(a+c)^{*}b\) \\
			\(R_{21}^2 \) & \( \emptyset +  (a+c)^{*} \emptyset\) & \(\emptyset\) \\
			\(R_{22}^2 \) & \( (a+c) + (a+c)^{*}\) & \((a+c)^{*}\) \\
			\(R_{23}^2 \) & \(  b + (a+c)^{*}b\) & \( (a+c)^{*}b\)     \\
			\(R_{31}^2 \) & \( \emptyset + \emptyset (a+c)^{*} \emptyset\) & \( \emptyset\) \\
			\(R_{32}^2 \) & \( c + c(a+c)^{*} \) & \( c(a+c)^{*} \)  \\
			\(R_{33}^2 \) & \( (a +b) + \emptyset (a+c)^{*}b \) & \( a + b \) \\
			\hline
		\end{tabular}
	\item \(k = 3\)
		\begin{tabular}[l|c|r]
			& direktes Einfügen & vereinfacht \\
			\hline \hline
			\(R_{13}^3 \) & \( (c^{*}(a+b)(a+c)^{*}b) + c^{*}(a+b)(a+c)^{*}b(a+b)^{*}  \) & \(c^{*}(a+b)								(a+c)^{*}b(a+b)^{*}\) \\
			\hline
		\end{tabular}
	\end{enumerate}
\end{homeworkProblem}
