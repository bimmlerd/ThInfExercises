\begin{homeworkProblem}
\begin{enumerate}[(a)]
\item Die Länge eines Programmes welches $0^{2^{2^{n^{5}}}}$ generiert ist nur von der Länge von $n$ abhängig, da der Rest immer gleich bleibt. Eine obere Schranke der Komplexität abhängig von $n$ ist deshalb
\begin{align}
K(w_n) \leq \ceil{\log_2{(n+1)}}+c' \leq \log_2{(n)}+c
\end{align}
mit zum Buch analoger Begründung. Nun soll die Schranke in Abhängigkeit zur Länge $\abs{w_n} = 2^{2^{(n^5)}}$ formuliert werden:
\begin{align*}
2^{2^{(n^5)}} &= \abs{w_n} \\
n^5 &= \log_2{\log_2{\abs{w_n}}} \\
n &= \sqrt[5]{\log_2{\log_2{\abs{w_n}}}} \\
\log_2{n} &= \log_2{\sqrt[5]{\log_2{\log_2{\abs{w_n}}}}} = \frac{1}{5}*\log_2{\log_2{\log_2{\abs{w_n}}}}
\end{align*}
Nun setzt man dieses Resultat in $(1)$ ein, und erhält:
\begin{align*}
K(w_n) \leq \frac{1}{5}*\log_2{\log_2{\log_2{\abs{w_n}}}}+c
\end{align*}

\item Eine solche Folge ist $y_i = 2^{((2^i)^3)}$. Es gilt $y_i<y_{i+1}$. Wir haben auch \begin{align*}
y_i &= 2^{((2^i)^3)} \\
\log_2{y_i} &= (2^i)^3 \\
\sqrt[3]{\log_2{y_i}} &= 2^i \\
i &= \log_2{\left(\sqrt[3]{\log_2{y_i}}\right)}
\end{align*}

Ein Programm, welches diese Folge generiert, ist in seiner Länge nur von $i$ abhängig. Mit der bekannten Kodierung von $i$ folgt:
\begin{align*}
K(y_i) &\le \ceil{\log_2{i}} + c \\
&= \ceil*{\log_2{\log_2{\left(\sqrt[3]{\log_2{y_i}}\right)}}} + c
\end{align*}

%\item Eine solche Folge ist $y_i = 2^{2^{\left({\left(2^{i}\right)}^{3}\right)}}$. Es gilt $y_i<y_{i+1}$. In der folgenden Implementation müssen wir einmal $i$ kodieren, alles andere ist konstant.
%
%\pascalscript{PascalImplementation}{Eine Pascal Implementation}

\end{enumerate}

\end{homeworkProblem}

\begin{homeworkProblem}
Wir führen einen Widerspruchsbeweis. Wir nehmen an, dass für mindestens die Hälfte der natürlichen Zahlen in $I = \;]0,2^n[$ gilt $K(i)<\ceil{\log_2{(2^n)}}-1$. 

In $I$ gibt es $\abs{[1,2^n-1]}$ viele Zahlen. Die Hälfte davon sind $\frac{\abs{I}}{2} = \frac{2^n-1}{2} = 2^{n-1} - \frac{1}{2}$. Da es \emph{mindestens} die Hälfte der Zahlen sein müssen, können wir auf \begin{align}
2^{n-1} - 0
\end{align} aufrunden.

Für $2^{n-1}$ Zahlen gilt $K(i)<\ceil{\log_2{(i+1)}}-1$. Es folgt direkt $K(i)\leq\ceil{\log_2{(i+1)}}-2$. Es stehen also $\ceil{\log_2{(i+1)}}-2$ Bits zur Darstellung dieser $2^{n-1}$ Zahlen zur Verfügung. Es gilt $i\leq2^n-1$, also können wir in der binären Darstellung \begin{align}
2^{\ceil{\log_2{(2^n-1+1)}}-2} = \frac{2^n}{4} = 2^{n-2}
\end{align} Zahlen darstellen.
Nun ist aber $(2) < (1)$, und wir haben einen Widerspruch. Es folgt, dass mindestens die Hälfte aller natürlichen Zahlen $i$ in $I$ die folgende Eigenschaft haben: \begin{align*}
K(i)\geq\ceil{\log_2{(i+1)}}-1
\end{align*}
\end{homeworkProblem}

\begin{homeworkProblem}
Ein Wort $x_n$ ist durch $i$ und $j$ eindeutig definiert. Zudem gilt $2(i+j) = \abs{x_n}$. Es folgt somit 
\begin{align*}
K(x_n) &\leq c + \log_2{i}+\log_2{j} \\
K(x_n) &\leq c + \log_2{(i*j)} \\
K(x_n) &\leq c + \log_2{(2*i*j)} \\
K(x_n) &\leq c + \log_2{(i^2 + 2ij + j^2)} \\
K(x_n) &\leq c + \log_2{(4i^2 + 8ij + 4j^2)} \\
K(x_n) &\leq c + \log_2{((2(i+j))^2)} \\
K(x_n) &\leq c + 2 * \log_2{(2(i+j))} \\
K(x_n) &\leq c + 2 * \log_2{(\abs{x_n})}
\end{align*}
und die Aussage ist bewiesen.
\end{homeworkProblem}
